Korišćenje Forme kontakt

From the sendemail.php file just replace the email address and save the file.

<?php

$name       = @trim(stripslashes($_POST['Ime'])); 

$from       = @trim(stripslashes($_POST['email'])); 

$subject    = @trim(stripslashes($_POST['naslov'])); 

$message    = @trim(stripslashes($_POST['poruka'])); 

$to    = 'email@email.com';//zameni sa svojim emailom


$headers   = array();

$headers[] = "MIME-Version: 1.0";

$headers[] = "Content-type: text/plain; charset=iso-8859-1";

$headers[] = "From: {$name} <{$from}>";

$headers[] = "Prosledi-na: <{$from}>";

$headers[] = "naslov: {$subject}";

$headers[] = "X-Mailer: PHP/".phpversion();


mail($to, $subject, $message, $headers);


die;
